## 阶段1：模板系统

### 1.1 为什么先做模板

OPC场景需要快速启动新业务线。模板是最佳实践的沉淀：
- 不需要从零配置
- 模板经过验证
- 新业务线可以基于模板

### 1.2 模板模型

```python
class TemplateType(str, Enum):
    BUSINESS_LINE = "business_line"
    WORKFLOW = "workflow"
    AGENT = "agent"

class BaseTemplate(BaseModel):
    """模板基类"""
    id: str
    name: str
    description: str
    template_type: TemplateType
    version: str = "1.0"
    created_at: float = Field(default_factory=time.time)

class AgentTemplate(BaseTemplate):
    """Agent模板"""
    template_type = TemplateType.AGENT
    name: str
    role: str
    system_prompt: str = ""
    capabilities: list[str] = []

class WorkflowTemplate(BaseTemplate):
    """Workflow模板"""
    template_type = TemplateType.WORKFLOW
    steps: list[WorkflowStep]
    evaluation_gates: list[str] = []  # 评估规则ID列表

class WorkflowStep(BaseModel):
    id: str
    agent_name: str
    task_description: str
    depends_on: list[str] = []  # 依赖的step id
    requires_approval: bool = False

class BusinessLineTemplate(BaseTemplate):
    """BusinessLine模板"""
    template_type = TemplateType.BUSINESS_LINE
    default_agents: list[AgentTemplate] = []
    default_tools: list[str] = []
    default_evaluation_rules: list[str] = []
    default_workflows: list[WorkflowTemplate] = []
    default_policies: list[str] = []  # Policy规则ID列表
```

### 1.4 模板注册表

创建 kernel/templates/registry.py：

```python
class TemplateRegistry:
    """模板注册表"""
    
    def __init__(self, storage: StorageBackend = None):
        self._storage = storage
        self._templates: dict[str, BaseTemplate] = {}
        
        # 加载内置模板
        for template in DEFAULT_TEMPLATES.values():
            self._templates[template.id] = template
    
    def register(self, template: BaseTemplate):
        """注册模板"""
        self._templates[template.id] = template
        if self._storage:
            self._storage.save(f"template:{template.id}", template)
    
    def get(self, template_id: str) -> BaseTemplate:
        """获取模板"""
        return self._templates.get(template_id)
    
    def list(self, template_type: TemplateType = None) -> list[BaseTemplate]:
        """列出模板"""
        templates = list(self._templates.values())
        if template_type:
            templates = [t for t in templates if t.template_type == template_type]
        return templates
    
    def delete(self, template_id: str):
        """删除模板（内置模板不能删除）"""
        if template_id in DEFAULT_TEMPLATES:
            raise ValueError(f"Cannot delete built-in template: {template_id}")
        self._templates.pop(template_id, None)
```

## 阶段2：Policy继承/覆盖

### 2.1 为什么需要继承/覆盖

OPC场景：
- Global Policy是全局安全底线
- 但不同业务线可能有不同需求
- 需要能覆盖Global Policy

### 2.2 Policy模型增强

修改 kernel/control/policy.py：

```python
class PolicyScope(str, Enum):
    GLOBAL = "global"      # 全局策略，所有业务线生效
    BUSINESS_LINE = "business_line"  # 业务线策略，覆盖Global

class PolicyRule(BaseModel):
    """策略规则"""
    id: str
    name: str
    scope: PolicyScope = PolicyScope.GLOBAL
    
    # 匹配条件
    condition: PolicyCondition
    
    # 执行动作
    action: PolicyAction
    
    # 效果
    effect: PolicyEffect  # ALLOW / DENY / REQUIRE_APPROVAL

class PolicyEngine:
    """策略引擎，支持继承和覆盖"""
    
    def __init__(self):
        self._global_policies: list[PolicyRule] = []
        self._bl_policies: dict[str, list[PolicyRule]] = {}  # bl_id -> policies
    
    def add_global_policy(self, rule: PolicyRule):
        """添加全局策略"""
        rule.scope = PolicyScope.GLOBAL
        self._global_policies.append(rule)
    
    def add_bl_policy(self, bl_id: str, rule: PolicyRule):
        """添加业务线策略（覆盖Global）"""
        rule.scope = PolicyScope.BUSINESS_LINE
        if bl_id not in self._bl_policies:
            self._bl_policies[bl_id] = []
        self._bl_policies[bl_id].append(rule)
    
    def evaluate(self, context: PolicyContext) -> PolicyEffect:
        """评估策略
        优先级：BusinessLine Policy > Global Policy
        """
        bl_id = context.business_line_id
        
        # 先检查BusinessLine Policy
        bl_policies = self._bl_policies.get(bl_id, [])
        for rule in bl_policies:
            if self._matches(rule, context):
                return rule.effect
        
        # 再检查Global Policy
        for rule in self._global_policies:
            if self._matches(rule, context):
                return rule.effect
        
        # 默认ALLOW
        return PolicyEffect.ALLOW
    
    def _matches(self, rule: PolicyRule, context: PolicyContext) -> bool:
        """检查规则是否匹配上下文"""
        condition = rule.condition
        
        if condition.type == "task_type":
            return context.task_type == condition.value
        elif condition.type == "tool_name":
            return context.tool_name == condition.value
        elif condition.type == "risk_level":
            return context.risk_level >= condition.value
        elif condition.type == "cost_exceeded":
            return context.cost_exceeded == condition.value
        
        return False
```

## 阶段3：成本预算 + HumanGate联动

### 3.1 为什么需要联动

V0.3的CostTracker已经能发布cost.budget_exceeded事件。
V0.4需要让这个事件触发HumanGate审批。

### 3.2 实现方式

修改 kernel/control/human_gate.py：

```python
class HumanApprovalGate:
    """人工审批门"""
    
    def __init__(
        self,
        notification_manager: NotificationManager = None,
        event_bus: EventBus = None,
        cost_tracker: CostTracker = None,
    ):
        self.notification_manager = notification_manager
        self.event_bus = event_bus
        self.cost_tracker = cost_tracker
        
        # 订阅成本超预算事件
        if self.event_bus:
            self.event_bus.subscribe("cost.budget_exceeded", self._on_cost_exceeded)
    
    async def _on_cost_exceeded(self, event: dict):
        """成本超预算事件处理"""
        provider = event.get("provider")
        budget = event.get("budget")
        current = event.get("current")
        
        # 创建审批任务
        approval_task = ApprovalTask(
            id=str(uuid.uuid4()),
            reason=f"成本超预算: {provider}当前${current:.2f}超过预算${budget:.2f}",
            task_type="cost_approval",
            priority="high",
            data={
                "provider": provider,
                "budget": budget,
                "current": current,
            }
        )
        
        # 加入审批队列
        self._pending_approvals.append(approval_task)
        
        # 通知
        if self.notification_manager:
            await self.notification_manager.notify_approval_required(
                task=None,
                reason=approval_task.reason
            )
    
    async def check_cost_and_approve(self, context: PolicyContext) -> bool:
        """检查成本并决定是否自动批准"""
        if not self.cost_tracker:
            return True
        
        provider = context.provider or "default"
        budget = self.cost_tracker._budgets.get(provider)
        
        if not budget:
            return True  # 没有设置预算，默认放行
        
        current_cost = self.cost_tracker._estimate_cost(provider)
        if current_cost > budget:
            return False  # 超预算，需要审批
        
        return True
```

### 4.1 跨Run记忆

```python
class PersistentContext:
    """持久化上下文，跨Run保留"""
    
    def __init__(self, storage: StorageBackend, business_line_id: str):
        self._storage = storage
        self._bl_id = business_line_id
        self._key_prefix = f"context:{business_line_id}:persistent:"
    
    async def save(self, key: str, value: Any):
        """保存到持久化上下文"""
        full_key = f"{self._key_prefix}{key}"
        await self._storage.save(full_key, value)
    
    async def load(self, key: str) -> Optional[Any]:
        """加载持久化上下文"""
        full_key = f"{self._key_prefix}{key}"
        return await self._storage.load(full_key)
    
    async def search(self, query: str) -> list[Any]:
        """搜索持久化上下文"""
        # 简化实现：返回所有
        prefix = self._key_prefix
        keys = await self._storage.list(prefix)
        return [await self._storage.load(k) for k in keys]
```

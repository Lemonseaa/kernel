```python
class BusinessLineStatus(str, Enum):
    """业务线状态。CREATE不是持久状态，创建完成后直接进入ACTIVE"""

    PAUSED = "paused"

    id: str = Field(default_factory=lambda: str(uuid.uuid4()))
    name: str
    status: BusinessLineStatus = BusinessLineStatus.ACTIVE
    config: BusinessLineConfig = BusinessLineConfig()
    created_at: float = Field(default_factory=time.time)
    last_active_at: float = Field(default_factory=time.time)

class BusinessLineConfig(BaseModel):
    """业务线配置"""
    evaluation_rules: list[str] = []
    resource_limits: ResourceLimits = ResourceLimits()

    """资源限制"""
```

## 阶段2：BusinessLineRegistry

创建 checkpoint_ai/businessline/registry.py：

```python
class BusinessLineRegistry:
    """业务线注册表"""
    
    def __init__(self, storage: StorageBackend):
        self._storage = storage
        self._cache: dict[str, BusinessLine] = {}
    
    def create(self, name: str, config: BusinessLineConfig = None) -> BusinessLine:
        """创建业务线，创建完成后直接进入ACTIVE"""
        bl = BusinessLine(name=name, config=config or BusinessLineConfig())
        key = f"business_line:{bl.id}"
        self._storage.save(key, bl)
        self._cache[bl.id] = bl
        return bl
    
    def get(self, bl_id: str) -> BusinessLine:
        """获取业务线"""
        if bl_id in self._cache:
            return self._cache[bl_id]
        key = f"business_line:{bl_id}"
        bl = self._storage.load(key)
        if bl:
            self._cache[bl_id] = bl
        return bl
    
    def list(self, status: BusinessLineStatus = None) -> list[BusinessLine]:
        """列出业务线"""
        prefix = "business_line:"
        keys = self._storage.list(prefix)
        business_lines = []
        for key in keys:
            bl = self._storage.load(key)
            if bl and (status is None or bl.status == status):
                business_lines.append(bl)
        return business_lines
    
    def update_status(self, bl_id: str, status: BusinessLineStatus) -> BusinessLine:
        """更新状态"""
        bl = self.get(bl_id)
        if bl is None:
            raise ValueError(f"BusinessLine not found: {bl_id}")
        bl.status = status
        bl.last_active_at = time.time()
        self._storage.save(f"business_line:{bl_id}", bl)
        self._cache[bl_id] = bl
        return bl
    
    def delete(self, bl_id: str):
        """删除业务线"""
        key = f"business_line:{bl_id}"
        self._storage.delete(key)
        self._cache.pop(bl_id, None)
```

```python
class AgentRegistry:
    """Agent注册表，按business_line_id隔离"""
    
    def register(self, profile: AgentProfile) -> Agent:
        """注册Agent"""
        key = f"agent:{profile.business_line_id}:{profile.id}"
        self._storage.save(key, agent)
        return agent
    
    def get(self, agent_id: str, business_line_id: str) -> Agent:
        """获取Agent，必须指定business_line_id"""
        key = f"agent:{business_line_id}:{agent_id}"
        return self._storage.load(key)
    
    def list(self, business_line_id: str) -> list[Agent]:
        """列出某业务线的所有Agent"""
        prefix = f"agent:{business_line_id}:"
        keys = self._storage.list(prefix)
        return [self._storage.load(k) for k in keys]
```

```python
def test_business_line_isolation():
    """验证两个业务线的资源完全隔离"""
    checkpoint_ai = CheckpointAI()
    
    bl_a = checkpoint_ai.create_business_line("A业务")
    bl_b = checkpoint_ai.create_business_line("B业务")
    
    # 在A业务注册Agent
    agent_a = bl_a.agents.register(AgentProfile(name="writer", business_line_id=bl_a.id))
    
    # 在B业务注册同名Agent
    agent_b = bl_b.agents.register(AgentProfile(name="writer", business_line_id=bl_b.id))
    
    # 验证隔离

    assert bl_b.agents.list() == [agent_b]
    
    # 验证上下文隔离
    await bl_a.context.save_result("writer", {"key": "A的数据"})
    await bl_b.context.save_result("writer", {"key": "B的数据"})
    
    assert await bl_a.context.load("writer") == {"key": "A的数据"}
    assert await bl_b.context.load("writer") == {"key": "B的数据"}
```

```python
class PluginRegistry:
    """插件注册表。只做注册和查询，不做动态加载/沙箱/热插拔"""
    
    def __init__(self):
        self._agents = {}
        self._tools = {}
        self._evaluators = {}
        self._providers = {}
    
    def register_agent(self, name: str, agent_class: type[BaseAgent]):
        self._agents[name] = agent_class
    
    def register_tool(self, name: str, tool_class: type[BaseTool]):
        self._tools[name] = tool_class
    
    def register_evaluator(self, name: str, evaluator_class: type[BaseEvaluator]):
        self._evaluators[name] = evaluator_class
    
    def register_provider(self, name: str, provider_class: type[LLMProvider]):
        self._providers[name] = provider_class
    
    def get_agent(self, name: str) -> type[BaseAgent]:
        return self._agents.get(name)
    
    def get_tool(self, name: str) -> type[BaseTool]:
        return self._tools.get(name)
    
    def get_evaluator(self, name: str) -> type[BaseEvaluator]:
        return self._evaluators.get(name)
    
    def get_provider(self, name: str) -> type[LLMProvider]:
        return self._providers.get(name)
    
    def list_agents(self) -> list[str]:
        return list(self._agents.keys())
    
    def list_tools(self) -> list[str]:
        return list(self._tools.keys())
    
    def list_evaluators(self) -> list[str]:
        return list(self._evaluators.keys())
    
    def list_providers(self) -> list[str]:
        return list(self._providers.keys())
```

## 阶段6：成本追踪作为Runtime Signal

创建 checkpoint_ai/observability/cost_tracker.py：

```python
class CostTracker:
    """成本追踪器。作为Runtime信号，接入事件流"""
    
    def __init__(self, event_bus: EventBus = None):
        self._counters: dict[str, TokenCounter] = {}
        self._event_bus = event_bus
        self._budgets: dict[str, float] = {}  # provider -> 预算上限
    
    def set_budget(self, provider: str, daily_budget: float):
        """设置每日预算"""
        self._budgets[provider] = daily_budget
    
    def track(self, provider: str, input_tokens: int, output_tokens: int):
        """记录Token消耗，发布成本事件"""
        if provider not in self._counters:
            self._counters[provider] = TokenCounter()
        
        self._counters[provider].input_tokens += input_tokens
        self._counters[provider].output_tokens += output_tokens
        
        # 发布成本事件，供Policy/HumanGate订阅
        if self._event_bus:
            self._event_bus.publish("cost.updated", {
                "provider": provider,
                "input_tokens": self._counters[provider].input_tokens,
                "output_tokens": self._counters[provider].output_tokens,
                "total_tokens": self._counters[provider].total_tokens,
            })
            
            # 检查是否超预算
            budget = self._budgets.get(provider)
            if budget and self._estimate_cost(provider) > budget:
                self._event_bus.publish("cost.budget_exceeded", {
                    "provider": provider,
                    "budget": budget,
                    "current": self._estimate_cost(provider),
                })
    
    def _estimate_cost(self, provider: str) -> float:
        """估算成本（简化版，后续接入PricingEngine）"""
        counter = self._counters.get(provider, TokenCounter())
        return counter.total_tokens * 0.001  # 简化估算
    
    def get_cost(self, provider: str = None) -> CostBreakdown:
        """获取成本"""
        if provider:
            counter = self._counters.get(provider, TokenCounter())
            return CostBreakdown(provider=provider, **vars(counter))
        
        # 返回所有provider的成本
        return {
            p: CostBreakdown(provider=p, **vars(c))
            for p, c in self._counters.items()
        }
    
    def reset(self, provider: str = None):
        """重置计数器"""
        if provider:
            self._counters[provider] = TokenCounter()
        else:
            self._counters.clear()

class TokenCounter:
    input_tokens: int = 0
    output_tokens: int = 0
    
    @property
    def total_tokens(self) -> int:
        return self.input_tokens + self.output_tokens

class CostBreakdown(BaseModel):
    provider: str
    input_tokens: int = 0
    output_tokens: int = 0
    total_tokens: int = 0
```

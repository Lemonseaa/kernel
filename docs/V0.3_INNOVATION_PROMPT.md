## 核心架构决策

**BusinessLine是第一等公民：**

- 不是tenant_id字段，是独立的实体
- 有完整生命周期：CREATE/ACTIVE/PAUSED/ARCHIVED/DELETED
- 业务线之间完全隔离
- Shared层资源可跨业务线复用

**高可扩展性：**

- 8个插件扩展点
- 新增功能通过注册机制，不是修改核心代码

class BusinessLine(BaseModel):

class BusinessLineConfig(BaseModel):

    resource_limits: ResourceLimits = ResourceLimits()

### 阶段4：资源按BusinessLine隔离

**原则：先让模型显式拥有business_line_id，再让查询层按它隔离。不是补丁式加字段，是模型层面的第一等属性。**

#### 4.2 Registry查询层隔离

```python
class AgentRegistry:
    """Agent注册表，按business_line_id隔离"""
    
    def __init__(self, storage: StorageBackend):
        self._storage = storage
    
    def register(self, profile: AgentProfile) -> Agent:
        """注册Agent，自动关联当前business_line_id"""
        key = f"agent:{profile.business_line_id}:{profile.id}"
        self._storage.save(key, agent)
        return agent
    
    def get(self, agent_id: str, business_line_id: str) -> Agent:
        """获取Agent，必须指定business_line_id"""
        key = f"agent:{business_line_id}:{agent_id}"
        return self._storage.load(key)
    
    def list(self, business_line_id: str) -> list[Agent]:
        """列出某业务线的所有Agent"""
        prefix = f"agent:{business_line_id}:"
        keys = self._storage.list(prefix)
        return [self._storage.load(k) for k in keys]

class ContextManager:
    """上下文管理，按business_line_id隔离"""
    
    async def get_context(self, agent_name: str, business_line_id: str) -> dict:
        """获取上下文，必须指定business_line_id"""
        key = f"context:{business_line_id}:{agent_name}"
        return await self._storage.load(key)
    
    async def save_result(self, agent_name: str, business_line_id: str, result: Any):
        """保存结果，按business_line_id隔离"""
        key = f"context:{business_line_id}:{agent_name}"
        await self._storage.save(key, result)

class RunStore:
    """Run存储，按business_line_id隔离"""
    
    def list_runs(self, business_line_id: str, status: RunStatus = None) -> list[Run]:
        """列出某业务线的所有Run"""
        prefix = f"run:{business_line_id}:"
        keys = self._storage.list(prefix)
        runs = [self._storage.load(k) for k in keys]
        if status:
            runs = [r for r in runs if r.status == status]
        return runs
```

#### 4.3 CheckpointAI层BusinessLine绑定

```python
class BusinessLine:
    """业务线封装，提供该业务线的隔离资源"""
    
    def __init__(self, bl_id: str, storage: StorageBackend):
        self.id = bl_id
        self._storage = storage
        self._agents = AgentRegistry(storage)
        self._context = ContextManager(storage)
        self._runs = RunStore(storage)
    
    @property
    def agents(self) -> AgentRegistry:
        """该业务线的Agent注册表"""
        return FilteredRegistry(self._agents, self.id)
    
    @property
    def context(self) -> ContextManager:
        """该业务线的上下文管理器"""
        return FilteredContextManager(self._context, self.id)
    
    def create_run(self, tasks: list[TaskSpec]) -> Run:
        """创建Run，自动带上business_line_id"""
        for task in tasks:
            task.business_line_id = self.id
        return self._runs.create(tasks)
```

#### 4.4 隔离验证测试

```python
async def test_business_line_isolation():
    """验证两个业务线的资源完全隔离"""
    checkpoint_ai = CheckpointAI()
    
    # 创建两个业务线
    bl_a = checkpoint_ai.create_business_line("A业务")
    bl_b = checkpoint_ai.create_business_line("B业务")
    
    # 在A业务注册Agent
    agent_a = bl_a.agents.register(AgentProfile(name="writer", business_line_id=bl_a.id))
    
    # 在B业务注册同名Agent
    agent_b = bl_b.agents.register(AgentProfile(name="writer", business_line_id=bl_b.id))
    
    # 验证隔离：A业务看不到B业务的Agent
    assert bl_a.agents.list() == [agent_a]
    assert bl_b.agents.list() == [agent_b]
    assert bl_a.agents.get(agent_b.id) is None
    
    # 验证上下文隔离
    await bl_a.context.save_result("writer", {"key": "A的数据"})
    await bl_b.context.save_result("writer", {"key": "B的数据"})
    
    assert await bl_a.context.load("writer") == {"key": "A的数据"}
    assert await bl_b.context.load("writer") == {"key": "B的数据"}
```

class PluginRegistry:
    """插件注册表"""
    
    def __init__(self):
        self._agents = {}
        self._tools = {}
        self._evaluators = {}
        self._providers = {}
    
    def register_agent(self, name: str, agent_class: type[BaseAgent]):
        self._agents[name] = agent_class
    
    def register_tool(self, name: str, tool_class: type[BaseTool]):
        self._tools[name] = tool_class
    
    def register_evaluator(self, name: str, evaluator_class: type[BaseEvaluator]):
        self._evaluators[name] = evaluator_class
    
    def get_agent(self, name: str) -> type[BaseAgent]:
        return self._agents.get(name)
    
    def get_tool(self, name: str) -> type[BaseTool]:
        return self._tools.get(name)
    
    def get_evaluator(self, name: str) -> type[BaseEvaluator]:

class CostTracker:
    """成本追踪"""
    
    def __init__(self):
        self._counters: dict[str, TokenCounter] = {}
    
    def track(self, provider: str, input_tokens: int, output_tokens: int):
        """记录Token消耗"""
    
    def get_cost(self, provider: str = None) -> CostBreakdown:
        """获取成本"""
    
    def reset(self, provider: str = None):
        """重置计数器"""

class TokenCounter:
    input_tokens: int = 0
    output_tokens: int = 0
    
    @property
    def total_tokens(self) -> int:
        return self.input_tokens + self.output_tokens
